import numpy as np

# 1. Ввод матрицы коэффициентов A и вектора b
A = np.array([, 
              [1, -2, 3], 
              [3, 4, -1]])
b = np.array()

# 2. Вычисление определителя (аналог Шага 2)
det_A = np.linalg.det(A)
print(f"Главный определитель: {det_A:.2f}")

# 3. Проверка условия Крамера
if abs(det_A) < 1e-9: # Учет погрешности float
    print("Определитель близок к 0, метод Крамера не применим.")
else:
    # 4. Решение системы (аналог Шага 4)
    # В реальных задачах ИИ используется solve, а не ручной Крамер
    X = np.linalg.solve(A, b)
